<template>

    <m-drag :list="list" :list2="list2">
      <template v-slot:list="items"> 
        <div class="div_box" v-for="(val,index) in items" :key="index">
          {{ val.label  }}
        </div>
      </template>  
      <template v-slot:list2="items"> 
        <div class="div_box" v-for="(val,index) in items" :key="index">
          {{ val.label  }}
        </div>
      </template>  
    </m-drag>

    <!--单列表-->

            </div>
        </template>  
    </m-drag>

      </template>  
    </m-drag>
</template>
